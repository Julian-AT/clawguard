```typescript
export const SeveritySchema = z.enum(["CRITICAL", "HIGH", "MEDIUM", "LOW", "INFO"]);
export const ConfidenceSchema = z.enum(["high", "medium", "low"]);
export const FindingSchema = z.object({
  severity: SeveritySchema, type: z.string(),
  location: z.object({ file: z.string(), line: z.number() }),
  cweId: z.string(), owaspCategory: z.string(),
  description: z.string(), attackScenario: z.string(),
  confidence: ConfidenceSchema,
  dataFlow: z.object({ source: z.string(), transform: z.string(), sink: z.string() }),
  fix: z.object({ before: z.string(), after: z.string() }),
  complianceMapping: z.object({ pciDss: z.string().optional(), soc2: z.string().optional(), hipaa: z.string().optional(), nist: z.string().optional(), owaspAsvs: z.string().optional() }),
});
export const PhaseResultSchema = z.object({ summary: z.string(), findings: z.array(FindingSchema) });
export const AuditResultSchema = z.object({
  phases: z.object({ quality: PhaseResultSchema, vulnerability: PhaseResultSchema, threatModel: PhaseResultSchema }),
  allFindings: z.array(FindingSchema),
  score: z.number(), grade: z.string(),
  severityCounts: z.record(z.string(), z.number()),
});
export type Finding = z.infer<typeof FindingSchema>;
export type PhaseResult = z.infer<typeof PhaseResultSchema>;
export type AuditResult = z.infer<typeof AuditResultSchema>;
```

    ```
        return result.output;
      } catch (error) {
        // On NoObjectGeneratedError or any error, return empty result
        console.error("[phase-X] Agent error:", error);
        return { summary: "Phase X analysis could not produce structured output", findings: [] };
      }
    }
    ```

    **1. Create `lib/analysis/pipeline.ts`:**

    Imports:
    ```
    import { Sandbox } from "@vercel/sandbox";
    import { createBashTool } from "bash-tool";
    import { runQualityReview } from "./phase1-quality";
    import { runVulnerabilityScan } from "./phase2-vuln";
    import { runThreatModel } from "./phase3-threat";
    import { calculateScore, countBySeverity } from "./scoring";
    import type { AuditResult, PhaseResult } from "./types";
    ```

    Export type `ProgressCallback = (phase: 'quality' | 'vulnerability' | 'threatModel', status: 'running' | 'complete' | 'error') => Promise<void>;`

    Export interface `PipelineInput`:
    ```
    { owner: string; repo: string; prBranch: string; baseBranch: string; }
    ```

    Export `async function runSecurityPipeline(input: PipelineInput, onProgress?: ProgressCallback): Promise<AuditResult>`:

    Implementation:
    - Create Sandbox with `timeout: 10 * 60 * 1000` (10 min, per research pitfall #3). Same git source pattern as current review.ts: `{ type: "git", url: "https://github.com/${input.owner}/${input.repo}", username: "x-access-token", password: process.env.GITHUB_TOKEN! }`
    - In try block:
      - `sandbox.runCommand("git", ["fetch", "origin", input.prBranch, input.baseBranch])`
      - `sandbox.runCommand("git", ["checkout", input.prBranch])`
      - Get diff: `sandbox.runCommand("git", ["diff", "origin/${input.baseBranch}...HEAD"])`, get stdout
      - Create bash tools: `const { tools } = await createBashTool({ sandbox })`
      - Phase 1: `await onProgress?.('quality', 'running')` then `const phase1 = await runQualityReview(tools, diff)` then `await onProgress?.('quality', 'complete')`
      - Phase 2: `await onProgress?.('vulnerability', 'running')` then `const phase2 = await runVulnerabilityScan(tools, diff, phase1.summary)` then `await onProgress?.('vulnerability', 'complete')`
      - Phase 3: `await onProgress?.('threatModel', 'running')` then `const phase3 = await runThreatModel(tools, diff, phase1.summary, phase2.summary)` then `await onProgress?.('threatModel', 'complete')`
      - Aggregate: `const allFindings = [...phase1.findings, ...phase2.findings, ...phase3.findings]`
      - Score: `const { score, grade } = calculateScore(allFindings)`
      - Counts: `const severityCounts = countBySeverity(allFindings)`
      - Return `{ phases: { quality: phase1, vulnerability: phase2, threatModel: phase3 }, allFindings, score, grade, severityCounts }`
    - In finally: `await sandbox.stop()`
